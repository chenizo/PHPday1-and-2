    private function setConn()
    {
        return
        is_null(self::$cnx)?
                self::$cnx=new PDO("mysql:host=localhost;dbname=DevPHP","root"):
                self::$cnx;
    }
    public function getConn()
    {
        return $this->setConn();
    }
